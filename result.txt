Here is the completed code for the task of picking an apple on the table and lifting it up for 5 cm using the left arm:

```python
from .base_task import Base_task
from .utils import *
import sapien

class pick_apple(Base_task):

    def setup_demo(self,**kwags): # load table and wall, setup robot and robot planner
        super()._init(**kwags)
        self.create_table_and_wall() 
        self.load_robot()
        self.setup_planner()
        self.load_camera(kwags.get('camera_w', 336),kwags.get('camera_h',224))
        self.pre_move()
        self.load_actors()

    def pre_move(self):
        pass

    def load_actors(self): # load apple
        self.apple, self.apple_data = create_glb(
            self.scene,
            pose=sapien.Pose([0, -0.06, 0.783],[0, 0, 0.995, 0.105]), # the z value (0.783) is important
            modelname="apple"
        )
        self.apple.find_component_by_type(sapien.physx.PhysxRigidDynamicComponent).mass = 0.001 # set apple mass

    def play_once(self):
        pose1 = self.get_grasp_pose_w_given_direction(self.apple, self.apple_data, grasp_qpos=[-0.5, 0.5, -0.5, -0.5], pre_dis=0.1) # apple pre grasp pose
        pose2 = self.get_grasp_pose_w_given_direction(self.apple, self.apple_data, grasp_qpos=[-0.5, 0.5, -0.5, -0.5]) # apple grasp pose
        
        self.open_left_gripper()  # open the left gripper first
        self.left_move_to_pose_with_screw(pose1) # left arm move to the pre grasp pose
        self.left_move_to_pose_with_screw(pose2) # left arm move to the grasp pose
        self.close_left_gripper()  # close left gripper to grasp the apple
        pose2[2] += 0.05  # lift the apple up for 5 cm
        self.left_move_to_pose_with_screw(pose2) # lift the apple up for 5 cm
        
    def check_success(self):
        apple_pos = self.apple.get_pose().p 
        apple_z = apple_pos[2]
        return apple_z - 0.783 > 0.05 # the apple is lifted up more than 5 cm
```

This code defines the `pick_apple` task, which involves loading an apple into the scene, picking it up using the left arm from a top-down approach, and lifting it 5 cm above its initial position. The `check_success` method verifies if the apple has been lifted by at least 5 cm.
